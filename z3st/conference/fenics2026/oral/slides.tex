\begin{frame}{The problem we keep meeting}
  Engineering components fail where \alert{heat, deformation and fracture meet}:
  nuclear fuel pellets, thermal-shock cracking, welds, additive parts.

  \vspace{0.6em}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Heavy multiphysics codes}\\(MOOSE, BISON)
      \begin{itemize}
        \item validated, production-grade
        \item large, hard to extend quickly
        \item weak-form changes are costly
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{FEniCSx}
      \begin{itemize}
        \item expressive weak forms in UFL
        \item symbolic differentiation built in
        \item but: no turnkey coupled\\ thermo-mechanical-fracture driver
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.6em}
  \begin{block}{Z3ST}
    A small, open framework that turns FEniCSx into a \alert{configuration-driven}
    solver for coupled thermo-mechanics, plasticity and phase-field fracture --
    with the constitutive physics written as \alert{energies}, not hand-coded tangents.
  \end{block}
\end{frame}

\begin{frame}[fragile]{The core idea: write the energy, get everything else}
  Constitutive models are entered as a \alert{strain-energy density}. UFL's
  automatic differentiation produces the stress and the consistent tangent.

  \vspace{0.4em}
  \begin{lstlisting}
F   = ufl.variable(I + grad(u))                  # deformation gradient
psi = (mu/2)*(Ic - 3) - mu*ufl.ln(J) \
      + (lmbda/2)*ufl.ln(J)**2                    # neo-Hookean energy

P   = ufl.diff(psi, F)                            # 1st Piola-Kirchhoff = d psi / d F
R   = ufl.inner(grad(v), P) * dx                  # residual
Jac = ufl.derivative(R, u, du)                    # consistent tangent -- also AD
  \end{lstlisting}

  \vspace{0.2em}
  \begin{itemize}
    \item one line, \texttt{P = ufl.diff(psi, F)}, replaces a hand-derived stress tensor
    \item the Newton tangent comes from \texttt{ufl.derivative} -- no analytic Jacobian
    \item one path for all: elasticity, hyperelasticity, J2 \alert{and crystal} plasticity, phase-field
    \item even slip-system crystal viscoplasticity gets its exact Jacobian from AD
  \end{itemize}
\end{frame}

\begin{frame}{Coupled thermo-mechanics, solved in a staggered loop}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      Per load step, the driver alternates until each field converges:
      \begin{enumerate}
        \item \textbf{thermal} $\;\rho c_p \dot T = \nabla\!\cdot(k(T)\nabla T) + q'''$
        \item \textbf{mechanical} $\;\nabla\!\cdot\sigma = 0$, with thermal eigenstrain\\
              $\varepsilon_{el} = \varepsilon(u) - \alpha\,(T-T_{ref})\,I$
        \item \textbf{damage} driven by $\psi^+(\varepsilon_{el})$, irreversible
      \end{enumerate}
      \vspace{0.5em}
      \alert{Adaptive relaxation}: per-field factors grow when the residual
      shrinks, shrink when it oscillates -- robust without a monolithic Newton.
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \begin{tikzpicture}[font=\scriptsize, node distance=5mm,
        s/.style={draw, rounded corners, minimum width=22mm, align=center, inner sep=3pt}]
        \node[s, fill=z3amber!20] (t) {thermal $T$};
        \node[s, fill=z3blue!12, below=of t] (m) {mechanical $u$};
        \node[s, fill=z3teal!15, below=of m] (d) {damage $D$};
        \draw[-{Stealth}] (t) -- (m);
        \draw[-{Stealth}] (m) -- (d);
        \draw[-{Stealth}, z3grey] (d.east) to[bend left=60] node[right=1pt]{until conv.} (t.east);
      \end{tikzpicture}
      \vspace{0.6em}

      {\scriptsize Damage degrades \alert{both} elastic and thermal-stress terms
      via a consistent $g(D)$ -- essential for tractable thermal-shock fracture.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Coupled multi-physics -- and across separate bodies}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Coupled physics in one solve}
      \begin{itemize}
        \item thermal $+$ mechanical $+$ damage / phase-field, two-way coupled
        \item temperature drives thermal strain and stress
        \item damage degrades both the elastic \alert{and} the thermal-stress terms
        \item phase-field: AT1 / AT2, Miehe / Amor, hybrid
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Heat across bodies: gap conductance}
      \begin{itemize}
        \item heat crosses a gap between \alert{separate bodies / regions}
        \item $h_{gap}(T_{gap})$ as a thermal interface condition
        \item e.g. fuel pellet $\to$ cladding -- essential for fuel-performance modelling
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.7em}
  \begin{block}{}
    \footnotesize
    \alert{New / exploratory:} cluster dynamics -- defect-population evolution in
    cluster-size space ($\partial_t c = -v\,\partial_n c + D\,\partial_{nn} c$).
    Implemented, not yet investigated; a path towards a genuine micro-to-continuum link.
  \end{block}
\end{frame}

\begin{frame}[fragile]{Backup: the staggered solver, in one screen}
\begin{lstlisting}
for iteration in range(max_iter):
    conv_th,   _ = self._thermal_step(...)       # k(T), q''' , transient
    conv_mech, _ = self._mechanical_step(...)    # eigenstrain, g(D)*sigma
    self.update_history(u_new, T=T_new)          # phase-field driver psi+
    conv_dmg,  _ = self._damage_step(...)        # irreversible D
    if conv_th and conv_mech and conv_dmg:
        return True                              # step converged
\end{lstlisting}
\vspace{0.3em}
{\scriptsize Adaptive per-field relaxation (grow / shrink on residual trend) keeps the
fixed-point iteration stable across linear elasticity through thermal-shock fracture.}
\end{frame}

\begin{frame}{Backup: burnup-driven PCMI -- a three-year rod history}
  \centering
  \includegraphics[width=0.95\textwidth]{figures/pcmi_burnup_curves.png}\\[2pt]
  \begin{itemize}\scriptsize
    \item burnup accumulates as a per-material state field; swelling is an
      eigenstrain callable that reads it -- \alert{``fuel is a material''},
      no solver change
    \item constant 25\,kW/m for $\sim$3 years: swelling closes the 65\,$\mu$m
      gap at $\sim$30\,MWd/kgU; the contact pressure \alert{emerges} and rises
      to $\sim$39\,MPa
    \item two-way feedback: contact raises the gap conductance, so the peak fuel
      temperature \alert{drops} as the gap closes (right panel)
  \end{itemize}
\end{frame}
